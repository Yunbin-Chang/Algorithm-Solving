# 풀이

로프의 최대 중률을 기준으로 내림 차순으로 정렬한 후, i번째의 i번째의 중량 x (i+1)을 통해 최대 중량을 구한다. 이렇게 구해진 중량이 전에 구한 중량보다 큰경우 최대 중량을 업데이트 한다.

``` c++
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;


int main(void) {
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);

    int N;
    cin >> N;

    vector<int> ropes(N);

    for (int i = 0; i < N; i++) {
        cin >> ropes[i];
    }

    sort(ropes.begin(), ropes.end(), greater<int>());

    long long max = 0;
    for (int i = 0; i < N; i++) {
        long long sum = ropes[i] * (i + 1);
        if (max < sum) {
            max = sum;
        }
    }

    cout << max;

    return 0;
```
